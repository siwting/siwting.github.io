查询并处理有退货的订单但实际是换货订单
SELECT * from db_order WHERE order_status=-2 ;


-- 对没有处理过的退货数据进行处理
update db_order o
LEFT JOIN db_order oo on o.cust_number=oo.cust_number and oo.order_id LIKE 'HT%' 
set o.handle_flag=-2
WHERE oo.order_id>'' and o.order_id LIKE 'HH%'  and  o.order_status!=-2   and o.handle_flag=0;



SELECT DISTINCT o.cust_number  from db_order o
LEFT JOIN db_order oo on o.cust_number=oo.cust_number and oo.order_id LIKE 'HT%' 
WHERE oo.order_id>'' and o.order_id LIKE 'HH%'  and  o.order_status!=-2  
GROUP BY oo.cust_number ;


-- 1、 查看当前客户编号是否符合要求
set @custNumber=(
SELECT DISTINCT o.cust_number  from db_order o
LEFT JOIN db_order oo on o.cust_number=oo.cust_number and oo.order_id LIKE 'HT%' 
WHERE oo.order_id>'' and o.order_id LIKE 'HH%'  and  o.order_status!=-2 
GROUP BY oo.cust_number LIMIT 1);

SELECT @custNumber;

SELECT order_id,xj_order_status,old_order_id,order_status,handle_flag,create_time,cust_number from db_order WHERE cust_number=@custNumber;


-- 2、 对符合要求的订单数据进行处理，并查看订单设备表的数据是否符合要求，符合要求，手动将订单设备状态修改为1
update db_order set order_status=1  WHERE cust_number=@custNumber and order_status=2 and  order_id not LIKE 'HH%';
update db_order set order_status=-2  WHERE cust_number=@custNumber and order_status=2 and  order_id LIKE 'HH%';


set @orderId=(
SELECT old_order_id from db_order WHERE cust_number=@custNumber and order_status=-2) ;

SELECT @orderId;
SELECT * from db_device_order WHERE order_id=@orderId;



-- 3、 对符合要求的设备客户数据进行处理，最后检查客户设备绑定关系是否正确，保证一个设备只有他对应的一个客户或是两个客户
set @devCode=(
SELECT dev_code from db_device_order WHERE order_id=@orderId and used_flag=1);
SELECT @devCode;

SELECT * from db_cust_device WHERE dev_code=@devCode;

UPDATE db_cust_device set bind_state=1 WHERE bind_state=0 and dev_code=@devCode and unbind_time is not null;
UPDATE db_cust_device set bind_state=0,update_uid=CONCAT("(无效)",update_uid) WHERE bind_state=1 and dev_code=@devCode and unbind_time is null;

SELECT u.real_name from db_customer u
LEFT JOIN db_cust_device ud on u.id=ud.cust_id and ud.bind_state=1
WHERE ud.dev_code>'' and ud.dev_code=@devCode;
