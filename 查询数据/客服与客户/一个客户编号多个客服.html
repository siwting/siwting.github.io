一个客户编号多个客服
SELECT COUNT(DISTINCT cuu.user_id)cnt,u.cust_number,GROUP_CONCAT(DISTINCT CONCAT("'",cuu.user_id,"'"))user_id
    from db_user_custom cuu
    LEFT JOIN db_cust_device ud on cuu.cust_id=ud.cust_id and ud.bind_state=1 and ud.cate_id=1
    LEFT JOIN db_customer u on u.id=ud.cust_id
    WHERE cuu.used_flag=1 and ud.dev_code>'' and u.cust_number>'' 
    GROUP BY u.cust_number
    HAVING cnt>1
