查询续费订单缺失的情况
SELECT o.* from db_order_details o
LEFT JOIN db_order_renew r on o.order_id=r.order_id 
WHERE r.id is null and o.product_no in('AJ010','AJ008');

-- 上面语句执行后，出现的3个订单的实际情况如下：
-- XJ1704204387 拒收
SELECT * from db_order WHERE order_id='XJ1704204387'

-- XJ1707011622 没有查到有我们的设备，但有续费订单
SELECT * from db_order WHERE order_id='XJ1707011622'
SELECT * from db_order WHERE cust_number='K16091700675';

SELECT * from db_device_order WHERE order_id='XJ1707011622'

-- XJ1707030628 没有查到有我们的设备，但有续费订单
SELECT * from db_order WHERE order_id='XJ1707030628';
SELECT * from db_order WHERE cust_number='K00000184105';
SELECT * from db_device_order WHERE order_id='XJ1707030628'
