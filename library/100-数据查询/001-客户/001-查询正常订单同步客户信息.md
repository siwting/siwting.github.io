```SQL
SELECT o.cust_number,o.cust_name,u.cust_name from db_order o
LEFT JOIN db_device_order doo on o.order_id=doo.order_id and doo.used_flag=1
LEFT JOIN db_cust_device ud on ud.dev_code=doo.dev_code and ud.bind_state=1
LEFT JOIN db_customer u on u.id=ud.cust_id
WHERE ud.dev_code>''  and o.order_time>'2017-07-01' and o.handle_flag>0  
and o.order_user_id <> '094C5A' and o.order_user_id<>'D49031' and o.order_status=1 and (u.cust_number is null or u.cust_number ='');
```

-- 处理语句 
```SQL
update  db_customer u
LEFT JOIN db_cust_device ud on u.id=ud.cust_id  and ud.bind_state=1
LEFT JOIN db_device_order doo on ud.dev_code=doo.dev_code and doo.used_flag=1
LEFT JOIN db_order o on o.order_id=doo.order_id and o.order_time>'2017-07-01' and o.handle_flag>0  
and o.order_user_id <> '094C5A' and o.order_user_id<>'D49031' and o.order_status=1
set u.cust_name=if(LOCATE("安佳",u.real_name)>0,o.cust_name,u.real_name),
		u.real_name=if(LOCATE("安佳",u.real_name)>0,o.cust_name,u.real_name),
		u.cust_phone=o.cust_phone,
		u.cust_phone2=o.cust_phone2,u.cust_phone3=o.cust_phone3,u.cust_phone4=o.cust_phone4,
		u.street= concat(o.province,o.city,o.county,o.detail_address),u.cust_number=o.cust_number,u.update_time=SYSDATE()
WHERE o.order_id>'' and (u.cust_number is null or u.cust_number ='');
```
